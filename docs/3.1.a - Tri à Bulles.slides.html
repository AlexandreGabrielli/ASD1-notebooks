def tri_a_bulles(T):
    N = len(T)
    for k in range(N,1,-1):     # N, N-1, ... 2
        for i in range(0,k-1):  # 0, 1, ... k-2 
            if T[i] > T[i+1]:   # T0 < T1 ... Tk-2 < Tk-1
                T[i],T[i+1] = T[i+1],T[i]   # swap
